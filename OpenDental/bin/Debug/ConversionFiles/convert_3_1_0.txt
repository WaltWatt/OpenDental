CREATE TABLE clearinghouse(
  ClearinghouseNum mediumint unsigned NOT NULL auto_increment,
  Description varchar(255) NOT NULL,
  ExportPath text NOT NULL,
  IsDefault tinyint(1) unsigned NOT NULL,
  Payors text NOT NULL,
  Eformat tinyint unsigned NOT NULL,
  ReceiverID varchar(255) NOT NULL,
  SenderID varchar(255) NOT NULL,
  Password varchar(255) NOT NULL,
  ResponsePath varchar(255) NOT NULL,
  CommBridge tinyint unsigned NOT NULL,
  ClientProgram varchar(255) NOT NULL,
  LastBatchNumber smallint unsigned NOT NULL,
  ModemPort tinyint unsigned NOT NULL,
  PRIMARY KEY (ClearinghouseNum)
);

CREATE TABLE providerident(
  ProviderIdentNum mediumint unsigned NOT NULL auto_increment,
  ProvNum smallint unsigned NOT NULL,
  PayorID varchar(255) NOT NULL,
  SuppIDType tinyint unsigned NOT NULL,
  IDNumber varchar(255) NOT NULL,
  PRIMARY KEY (ProviderIdentNum)
);

CREATE TABLE electid(
  ElectIDNum mediumint unsigned NOT NULL auto_increment,
  PayorID varchar(255) NOT NULL,
  CarrierName varchar(255) NOT NULL,
  IsMedicaid tinyint unsigned NOT NULL,
  ProviderTypes varchar(255) NOT NULL,
  Comments text NOT NULL,
  PRIMARY KEY (ElectIDNum)
);

CREATE TABLE recall(
  RecallNum mediumint unsigned NOT NULL auto_increment,
  PatNum mediumint unsigned NOT NULL,
  DateDueCalc date NOT NULL,
  DateDue date NOT NULL,
  DatePrevious date NOT NULL,
  RecallInterval int NOT NULL,
  RecallStatus smallint unsigned NOT NULL,
  Note text NOT NULL,
  IsDisabled tinyint unsigned NOT NULL,
  PRIMARY KEY (RecallNum),
  INDEX (PatNum),
  INDEX (DateDue)
);
